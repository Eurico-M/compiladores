with Ada.Text_IO; use Ada.Text_IO;

procedure Main is
   X : Integer := 42;
   Y : String := "Hello";
   Z : Boolean := false;
begin
   Get_Line(X);
   Put_Line(X % 3);

   -- if-then statement
   if X = 0 then
      Put_Line(X + 1);
      X := 5;
   end if;

   -- if-then-else statement
   if X > 3 then
      Put_Line(100);   -- Example action if X > 3
   else
      Put_Line(200);   -- Example action if X <= 3
   end if;

   -- While loop
   while X > 0 loop
      Put_Line(X - 1);
      Put_Line(X / 7);
      X := X - 1;
   end loop;

   -- Although left associative, multiplication and division have priority
   Put_Line(1 + X * X / 3);
end Main;
